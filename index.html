        <label for="selling">Selling Price:</label>

        <input type="number" id="selling" placeholder="Enter selling price">

        <button onclick="calculateMargin()">Calculate Margin</button>

        <div class="result" id="result"></div>

    </div>

    <script>

        function calculateMargin() {

            const cost = parseFloat(document.getElementById('cost').value);

            const selling = parseFloat(document.getElementById('selling').value);

            if (!isNaN(cost) && !isNaN(selling) && cost > 0 && selling > 0) {

                const margin = ((selling - cost) / selling) * 100;

                document.getElementById('result').innerHTML = `Margin: ${margin.toFixed(2)}%`;

            } else {

                document.getElementById('result').innerHTML = 'Please enter valid values';

            }

        }

    </script>
